冒泡排序

```python
def bubble_sort(list):
    #记录一个索引，这个索引右边的元素都已经正确排序
    unsorted_until_index = len(list)-1
    sorted = False
    while not sorted:
        # 位置不再交换则排序已经完成
        sorted = True
        # 这里要从开头一直移动到尚未排序的索引
        for i in range(unsorted_until_index):
            # 判断条件，如果此位置大于下一个位置，就需要将二者交换位置
            if list[i] > list[i+1]:
                list[i] ,list[i+1] = list[i+1],list[i]
                sorted = False
        # 排序好一个，未排序的总数就减少一个
        unsorted_until_index -= 1
    return list
```

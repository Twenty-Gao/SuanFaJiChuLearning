def insertion_sort(array):
    # 需要插入的数据的位置
    # 首先使用一个循环，从索引1处开始遍历数组，每一轮循环都代表一次遍历
    for index in range(1,len(array)):
        # 每次遍历中，把”移除“的值存入变量 temp_value中
